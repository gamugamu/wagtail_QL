    retourne tout les pma de la page home

    query pma{ allPmaHome{ title caption id dateStart dateEnd } }

    retourne toutes les galleries de la page home

    query gallery{ allPmaGallery{ title caption id dateStart dateEnd } }

    Mutations

    modifie une page home

    mutation myMutation { mutatePmaHome(pmaData: {title: "new pma", caption: "lobulous"}) { pma{ title caption } } }

    modifie une gallerie

    mutation myMutation { mutatePmaGallery(pmaData: { id: 1, title: "new pma from js", caption: "lobulous", gallery: [ {title:"Deux titre"}]}) { pma{ title, caption, gallery{ title caption } } } }
